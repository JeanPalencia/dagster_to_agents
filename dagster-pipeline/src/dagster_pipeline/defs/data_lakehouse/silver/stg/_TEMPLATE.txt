@dg.asset(
    partitions_def=daily_partitions,
    group_name="silver",
    description="Silver STG: {TABLE_NAME} from Spot2 Platform.",
    io_manager_key="s3_silver",
)
def stg_s2p_{TABLE_NAME}_new(
    context: dg.AssetExecutionContext,
    {BRONZE_ASSET}: pl.DataFrame,
) -> pl.DataFrame:
    """Transform {BRONZE_ASSET}; validations run before/after transform."""
    input_validation = validate_input_bronze({BRONZE_ASSET}, context)

    if input_validation["is_empty"]:
        context.log.warning("⚠️  Empty input — returning empty DataFrame")
        return {BRONZE_ASSET}

    df_transformed = _transform_s2p_{TABLE_NAME}({BRONZE_ASSET})

    context.log.info(
        f"stg_s2p_{TABLE_NAME}_new: {df_transformed.height} rows for partition {context.partition_key}"
    )

    output_validation = validate_output_silver(
        df_transformed,
        {BRONZE_ASSET},
        context,
        allow_row_loss=False,
    )

    if not output_validation["validation_passed"]:
        context.log.error(
            f"❌ Validation failed: {output_validation['rows_lost']:,} rows lost. Check transforms."
        )

    df_transformed = validate_and_clean_silver(df_transformed, context)

    s3_key = build_silver_s3_key(f"stg_s2p_{TABLE_NAME}_new", file_format="parquet")
    write_polars_to_s3(df_transformed, s3_key, context, file_format="parquet")

    return df_transformed
